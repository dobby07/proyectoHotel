    private void MostrarTabla(){
        ResultSet rs;
        clsNCliente objNclie = new clsNCliente();
        DefaultTableModel model = new DefaultTableModel();
        Object[] dato = new Object[6]; // Cambiamos el tipo de dato a Object

        try {
            rs = objNclie.Mtdlistar();
            model.setColumnIdentifiers(new Object[] {"DNI", "NOMBRE","APELLIDO","TELEFONO","Editar", "Eliminar"}); // Agregamos los nombres de columna al modelo
            while (rs.next()) {
                dato[0] = rs.getString("dni");
                dato[1] = rs.getString("nombre");
                dato[2] = rs.getString("apellido");
                dato[3] = rs.getString("telefono");
                dato[4] = new JButton("Editar");
                dato[5] = new JButton("Eliminar");// Agregamos el botón en la última posición
                model.addRow(dato);
            }
            this.TABLACLIENTE.setModel(model);
            this.TABLACLIENTE.setDefaultRenderer(Object.class, new RenderTabla());
            
            //damos altura y ancho a las columnas
            this.TABLACLIENTE.setRowHeight(40);
            
        } catch (Exception ex) {
            System.out.println(ex.getMessage());
        }

    }
    
    private void btnListarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnListarActionPerformed

        MostrarTabla();

        // Agrega el ListSelectionListener a la JTable
        TABLACLIENTE.getSelectionModel().addListSelectionListener((ListSelectionEvent e) -> {
            // Verificar si la selección aún está cambiando y si hay una fila seleccionada
            if (!e.getValueIsAdjusting() && TABLACLIENTE.getSelectedRow() != -1) {
                // Obtener los valores de la fila seleccionada
                String dni = TABLACLIENTE.getValueAt(TABLACLIENTE.getSelectedRow(), 0).toString();
                String direccion = TABLACLIENTE.getValueAt(TABLACLIENTE.getSelectedRow(), 1).toString();
                String telefono = TABLACLIENTE.getValueAt(TABLACLIENTE.getSelectedRow(), 2).toString();
                String email = TABLACLIENTE.getValueAt(TABLACLIENTE.getSelectedRow(), 3).toString();

                // Asignar los valores a las cajas de texto correspondientes
                txtdni.setText(dni);
                txtnombre.setText(direccion);
                txtapellido.setText(telefono);
                txttelefono.setText(email);
            }
        });

    private void btnBuscarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnBuscarActionPerformed

        clsECliente objEcli = new clsECliente();
        clsNCliente objNcli = new clsNCliente();

        String codBuscar = txtCodbuscar.getText().trim(); // Obtener el valor de búsqueda y eliminar espacios en blanco

        // Verificar si el campo de búsqueda está vacío
        if (codBuscar.isEmpty()) {
            MostrarTabla(); // Llama a la función MostrarTabla() para listar todos los datos
        } else {
            // Ejecutar la búsqueda según el código ingresado
            objEcli.setDni(txtdni.getText());
            ResultSet rsc = objNcli.MtdBuscarCodigo(objEcli);
            DefaultTableModel model = (DefaultTableModel) TABLACLIENTE.getModel();
            model.setRowCount(0); // Limpiar el modelo de la tabla

            try {
                if (rsc.next()) {
                    Object[] rowData = new Object[6];
                    rowData[0] = rsc.getString("DNI");
                    rowData[1] = rsc.getString("NOMBRE");
                    rowData[2] = rsc.getString("APELLIDO");
                    rowData[3] = rsc.getString("TELEFONO");
                    rowData[4] = new JButton("Editar");
                    rowData[5] = new JButton("Eliminar");

                    model.addRow(rowData);
                } else {
                    JOptionPane.showMessageDialog(null, "No se encontraron resultados para el código ingresado.");
                }
            } catch (Exception ex) {
                System.out.println(ex.getMessage());
            }
        }
        // TODO add your handling code here:
    }//GEN-LAST:event_btnBuscarActionPerformed

    private void btnGuardarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnGuardarActionPerformed

        clsECliente objEclie=new clsECliente();
        clsNCliente objNclie=new clsNCliente();

        objEclie.setDni(txtdni.getText());
        objEclie.setNombre(txtnombre.getText());
        objEclie.setApellido(txtapellido.getText());
        objEclie.setTelefono(txttelefono.getText());

        if(objNclie.MtdAgregar(objEclie)){
            JOptionPane.showMessageDialog(null,"Datos Guardados");
            MostrarTabla();
            Limpiar();
        }
        else{
            JOptionPane.showMessageDialog(null,"Datos Erroneos");
        }

    private void TABLACLIENTEMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_TABLACLIENTEMouseClicked
        int fila = TABLACLIENTE.getSelectedRow();
        int columna = TABLACLIENTE.getSelectedColumn();

        if (columna == 4) { // Verificar si se hizo clic en la columna del botón "Editar"
            JButton botonEditar = (JButton) TABLACLIENTE.getValueAt(fila, columna);

            // Acciones a realizar cuando se hace clic en el botón "Editar"
            if (botonEditar != null && evt.getClickCount() == 1) {
                String dni = (String) TABLACLIENTE.getValueAt(fila, 0);
                String direccion = (String) TABLACLIENTE.getValueAt(fila, 1);
                String telefono = (String) TABLACLIENTE.getValueAt(fila, 2);
                String email = (String) TABLACLIENTE.getValueAt(fila, 3);

                FrmClienteInternal formulario = new FrmClienteInternal();
                formulario.setDatos(dni,direccion,telefono,email);
                formulario.setVisible(true);
            }
        }
        else if(columna == 5){
            // Verificar si se hizo clic en la columna del botón "Eliminar"
            JButton botonEliminar = (JButton) TABLACLIENTE.getValueAt(fila, columna);

            // Acciones a realizar cuando se hace clic en el botón "Eliminar"
            if (botonEliminar != null && evt.getClickCount() == 1) {
                int confirmacion = JOptionPane.showConfirmDialog(null, "¿Estás seguro de eliminar este registro?", "Confirmar Eliminación", JOptionPane.YES_NO_OPTION);
                if (confirmacion == JOptionPane.YES_OPTION) {
                    String dni = (String) TABLACLIENTE.getValueAt(fila, 0);

                    // Lógica para eliminar el registro de la base de datos
                    clsNCliente objNclie = new clsNCliente();
                    if (objNclie.MtdEliminar(dni)) {
                        DefaultTableModel model = (DefaultTableModel) TABLACLIENTE.getModel();
                        model.removeRow(fila);
                        JOptionPane.showMessageDialog(null, "Registro eliminado correctamente");
                    } else {
                        JOptionPane.showMessageDialog(null, "Error al eliminar el registro");
                    }
                }
            }
        }

        // TODO add your handling code here:
    }//GEN-LAST:event_TABLACLIENTEMouseClicked
